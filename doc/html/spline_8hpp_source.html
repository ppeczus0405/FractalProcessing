55// spline interpolation
56class spline
57{
58public:
59 // spline types
60 enum spline_type {
61 linear = 10, // linear interpolation
62 cspline = 30, // cubic splines (classical C^2)
63 cspline_hermite = 31 // cubic hermite splines (local, only C^1)
64 };
65
66 // boundary condition type for the spline end-points
67 enum bd_type {
68 first_deriv = 1,
69 second_deriv = 2
70 };
71
72protected:
73 std::vector<double> m_x,m_y; // x,y coordinates of points
74 // interpolation parameters
75 // f(x) = a_i + b_i*(x-x_i) + c_i*(x-x_i)^2 + d_i*(x-x_i)^3
76 // where a_i = y_i, or else it won't go through grid points
77 std::vector<double> m_b,m_c,m_d; // spline coefficients
78 double m_c0; // for left extrapolation
79 spline_type m_type;
80 bd_type m_left, m_right;
81 double m_left_value, m_right_value;
82 bool m_made_monotonic;
83 void set_coeffs_from_b(); // calculate c_i, d_i from b_i
84 size_t find_closest(double x) const; // closest idx so that m_x[idx]<=x
85
86public:
87 // default constructor: set boundary condition to be zero curvature
88 // at both ends, i.e. natural splines
89 spline(): m_type(cspline),
90 m_left(second_deriv), m_right(second_deriv),
91 m_left_value(0.0), m_right_value(0.0), m_made_monotonic(false)
92 {
93 ;
94 }
95 spline(const std::vector<double>& X, const std::vector<double>& Y,
96 spline_type type = cspline,
97 bool make_monotonic = false,
98 bd_type left = second_deriv, double left_value = 0.0,
99 bd_type right = second_deriv, double right_value = 0.0
100 ):
101 m_type(type),
102 m_left(left), m_right(right),
103 m_left_value(left_value), m_right_value(right_value),
104 m_made_monotonic(false) // false correct here: make_monotonic() sets it
105 {
106 this->set_points(X,Y,m_type);
107 if(make_monotonic) {
108 this->make_monotonic();
109 }
110 }
111
112
113 // modify boundary conditions: if called it must be before set_points()
114 void set_boundary(bd_type left, double left_value,
115 bd_type right, double right_value);
116
117 // set all data points (cubic_spline=false means linear interpolation)
118 void set_points(const std::vector<double>& x,
119 const std::vector<double>& y,
120 spline_type type=cspline);
121
122 // adjust coefficients so that the spline becomes piecewise monotonic
123 // where possible
124 // this is done by adjusting slopes at grid points by a non-negative
125 // factor and this will break C^2
126 // this can also break boundary conditions if adjustments need to
127 // be made at the boundary points
128 // returns false if no adjustments have been made, true otherwise
129 bool make_monotonic();
130
131 // evaluates the spline at point x
132 double operator() (double x) const;
133 double deriv(int order, double x) const;
134
135 // returns the input data points
136 std::vector<double> get_x() const { return m_x; }
137 std::vector<double> get_y() const { return m_y; }
138 double get_x_min() const { assert(!m_x.empty()); return m_x.front(); }
139 double get_x_max() const { assert(!m_x.empty()); return m_x.back(); }
140
141#ifdef HAVE_SSTREAM
142 // spline info string, i.e. spline type, boundary conditions etc.
143 std::string info() const;
144#endif // HAVE_SSTREAM
145
146};

210void spline::set_coeffs_from_b()
211{
212 assert(m_x.size()==m_y.size());
213 assert(m_x.size()==m_b.size());
214 assert(m_x.size()>2);
215 size_t n=m_b.size();
216 if(m_c.size()!=n)
217 m_c.resize(n);
218 if(m_d.size()!=n)
219 m_d.resize(n);
220
221 for(size_t i=0; i<n-1; i++) {
222 const double h = m_x[i+1]-m_x[i];
223 // from continuity and differentiability condition
224 m_c[i] = ( 3.0*(m_y[i+1]-m_y[i])/h - (2.0*m_b[i]+m_b[i+1]) ) / h;
225 // from differentiability condition
226 m_d[i] = ( (m_b[i+1]-m_b[i])/(3.0*h) - 2.0/3.0*m_c[i] ) / h;
227 }
228
229 // for left extrapolation coefficients
230 m_c0 = (m_left==first_deriv) ? 0.0 : m_c[0];
231}

233void spline::set_points(const std::vector<double>& x,
234 const std::vector<double>& y,
235 spline_type type)
236{
237 assert(x.size()==y.size());
238 assert(x.size()>2);
239 m_type=type;
240 m_made_monotonic=false;
241 m_x=x;
242 m_y=y;
243 int n = (int) x.size();
244 // check strict monotonicity of input vector x
245 for(int i=0; i<n-1; i++) {
246 assert(m_x[i]<m_x[i+1]);
247 }
248
249
250 if(type==linear) {
251 // linear interpolation
252 m_d.resize(n);
253 m_c.resize(n);
254 m_b.resize(n);
255 for(int i=0; i<n-1; i++) {
256 m_d[i]=0.0;
257 m_c[i]=0.0;
258 m_b[i]=(m_y[i+1]-m_y[i])/(m_x[i+1]-m_x[i]);
259 }
260 // ignore boundary conditions, set slope equal to the last segment
261 m_b[n-1]=m_b[n-2];
262 m_c[n-1]=0.0;
263 m_d[n-1]=0.0;
264 } else if(type==cspline) {
265 // classical cubic splines which are C^2 (twice cont differentiable)
266 // this requires solving an equation system
267
268 // setting up the matrix and right hand side of the equation system
269 // for the parameters b[]
270 internal::band_matrix A(n,1,1);
271 std::vector<double> rhs(n);
272 for(int i=1; i<n-1; i++) {
273 A(i,i-1)=1.0/3.0*(x[i]-x[i-1]);
274 A(i,i)=2.0/3.0*(x[i+1]-x[i-1]);
275 A(i,i+1)=1.0/3.0*(x[i+1]-x[i]);
276 rhs[i]=(y[i+1]-y[i])/(x[i+1]-x[i]) - (y[i]-y[i-1])/(x[i]-x[i-1]);
277 }
278 // boundary conditions
279 if(m_left == spline::second_deriv) {
280 // 2*c[0] = f''
281 A(0,0)=2.0;
282 A(0,1)=0.0;
283 rhs[0]=m_left_value;
284 } else if(m_left == spline::first_deriv) {
285 // b[0] = f', needs to be re-expressed in terms of c:
286 // (2c[0]+c[1])(x[1]-x[0]) = 3 ((y[1]-y[0])/(x[1]-x[0]) - f')
287 A(0,0)=2.0*(x[1]-x[0]);
288 A(0,1)=1.0*(x[1]-x[0]);
289 rhs[0]=3.0*((y[1]-y[0])/(x[1]-x[0])-m_left_value);
290 } else {
291 assert(false);
292 }
293 if(m_right == spline::second_deriv) {
294 // 2*c[n-1] = f''
295 A(n-1,n-1)=2.0;
296 A(n-1,n-2)=0.0;
297 rhs[n-1]=m_right_value;
298 } else if(m_right == spline::first_deriv) {
299 // b[n-1] = f', needs to be re-expressed in terms of c:
300 // (c[n-2]+2c[n-1])(x[n-1]-x[n-2])
301 // = 3 (f' - (y[n-1]-y[n-2])/(x[n-1]-x[n-2]))
302 A(n-1,n-1)=2.0*(x[n-1]-x[n-2]);
303 A(n-1,n-2)=1.0*(x[n-1]-x[n-2]);
304 rhs[n-1]=3.0*(m_right_value-(y[n-1]-y[n-2])/(x[n-1]-x[n-2]));
305 } else {
306 assert(false);
307 }
308
309 // solve the equation system to obtain the parameters c[]
310 m_c=A.lu_solve(rhs);
311
312 // calculate parameters b[] and d[] based on c[]
313 m_d.resize(n);
314 m_b.resize(n);
315 for(int i=0; i<n-1; i++) {
316 m_d[i]=1.0/3.0*(m_c[i+1]-m_c[i])/(x[i+1]-x[i]);
317 m_b[i]=(y[i+1]-y[i])/(x[i+1]-x[i])
318 - 1.0/3.0*(2.0*m_c[i]+m_c[i+1])*(x[i+1]-x[i]);
319 }
320 // for the right extrapolation coefficients (zero cubic term)
321 // f_{n-1}(x) = y_{n-1} + b*(x-x_{n-1}) + c*(x-x_{n-1})^2
322 double h=x[n-1]-x[n-2];
323 // m_c[n-1] is determined by the boundary condition
324 m_d[n-1]=0.0;
325 m_b[n-1]=3.0*m_d[n-2]*h*h+2.0*m_c[n-2]*h+m_b[n-2]; // = f'_{n-2}(x_{n-1})
326 if(m_right==first_deriv)
327 m_c[n-1]=0.0; // force linear extrapolation
328
329 } else if(type==cspline_hermite) {
330 // hermite cubic splines which are C^1 (cont. differentiable)
331 // and derivatives are specified on each grid point
332 // (here we use 3-point finite differences)
333 m_b.resize(n);
334 m_c.resize(n);
335 m_d.resize(n);
336 // set b to match 1st order derivative finite difference
337 for(int i=1; i<n-1; i++) {
338 const double h = m_x[i+1]-m_x[i];
339 const double hl = m_x[i]-m_x[i-1];
340 m_b[i] = -h/(hl*(hl+h))*m_y[i-1] + (h-hl)/(hl*h)*m_y[i]
341 + hl/(h*(hl+h))*m_y[i+1];
342 }
343 // boundary conditions determine b[0] and b[n-1]
344 if(m_left==first_deriv) {
345 m_b[0]=m_left_value;
346 } else if(m_left==second_deriv) {
347 const double h = m_x[1]-m_x[0];
348 m_b[0]=0.5*(-m_b[1]-0.5*m_left_value*h+3.0*(m_y[1]-m_y[0])/h);
349 } else {
350 assert(false);
351 }
352 if(m_right==first_deriv) {
353 m_b[n-1]=m_right_value;
354 m_c[n-1]=0.0;
355 } else if(m_right==second_deriv) {
356 const double h = m_x[n-1]-m_x[n-2];
357 m_b[n-1]=0.5*(-m_b[n-2]+0.5*m_right_value*h+3.0*(m_y[n-1]-m_y[n-2])/h);
358 m_c[n-1]=0.5*m_right_value;
359 } else {
360 assert(false);
361 }
362 m_d[n-1]=0.0;
363
364 // parameters c and d are determined by continuity and differentiability
365 set_coeffs_from_b();
366
367 } else {
368 assert(false);
369 }
370
371 // for left extrapolation coefficients
372 m_c0 = (m_left==first_deriv) ? 0.0 : m_c[0];
373}

375bool spline::make_monotonic()
376{
377 assert(m_x.size()==m_y.size());
378 assert(m_x.size()==m_b.size());
379 assert(m_x.size()>2);
380 bool modified = false;
381 const int n=(int)m_x.size();
382 // make sure: input data monotonic increasing --> b_i>=0
383 // input data monotonic decreasing --> b_i<=0
384 for(int i=0; i<n; i++) {
385 int im1 = std::max(i-1, 0);
386 int ip1 = std::min(i+1, n-1);
387 if( ((m_y[im1]<=m_y[i]) && (m_y[i]<=m_y[ip1]) && m_b[i]<0.0) ||
388 ((m_y[im1]>=m_y[i]) && (m_y[i]>=m_y[ip1]) && m_b[i]>0.0) ) {
389 modified=true;
390 m_b[i]=0.0;
391 }
392 }
393 // if input data is monotonic (b[i], b[i+1], avg have all the same sign)
394 // ensure a sufficient criteria for monotonicity is satisfied:
395 // sqrt(b[i]^2+b[i+1]^2) <= 3 |avg|, with avg=(y[i+1]-y[i])/h,
396 for(int i=0; i<n-1; i++) {
397 double h = m_x[i+1]-m_x[i];
398 double avg = (m_y[i+1]-m_y[i])/h;
399 if( avg==0.0 && (m_b[i]!=0.0 || m_b[i+1]!=0.0) ) {
400 modified=true;
401 m_b[i]=0.0;
402 m_b[i+1]=0.0;
403 } else if( (m_b[i]>=0.0 && m_b[i+1]>=0.0 && avg>0.0) ||
404 (m_b[i]<=0.0 && m_b[i+1]<=0.0 && avg<0.0) ) {
405 // input data is monotonic
406 double r = sqrt(m_b[i]*m_b[i]+m_b[i+1]*m_b[i+1])/std::fabs(avg);
407 if(r>3.0) {
408 // sufficient criteria for monotonicity: r<=3
409 // adjust b[i] and b[i+1]
410 modified=true;
411 m_b[i] *= (3.0/r);
412 m_b[i+1] *= (3.0/r);
413 }
414 }
415 }
416
417 if(modified==true) {
418 set_coeffs_from_b();
419 m_made_monotonic=true;
420 }
421
422 return modified;
423}

425// return the closest idx so that m_x[idx] <= x (return 0 if x<m_x[0])
426size_t spline::find_closest(double x) const
427{
428 std::vector<double>::const_iterator it;
429 it=std::upper_bound(m_x.begin(),m_x.end(),x); // *it > x
430 size_t idx = std::max( int(it-m_x.begin())-1, 0); // m_x[idx] <= x
431 return idx;
432}

434double spline::operator() (double x) const
435{
436 // polynomial evaluation using Horner's scheme
437 // TODO: consider more numerically accurate algorithms, e.g.:
438 // - Clenshaw
439 // - Even-Odd method by A.C.R. Newbery
440 // - Compensated Horner Scheme
441 size_t n=m_x.size();
442 size_t idx=find_closest(x);
443
444 double h=x-m_x[idx];
445 double interpol;
446 if(x<m_x[0]) {
447 // extrapolation to the left
448 interpol=(m_c0*h + m_b[0])*h + m_y[0];
449 } else if(x>m_x[n-1]) {
450 // extrapolation to the right
451 interpol=(m_c[n-1]*h + m_b[n-1])*h + m_y[n-1];
452 } else {
453 // interpolation
454 interpol=((m_d[idx]*h + m_c[idx])*h + m_b[idx])*h + m_y[idx];
455 }
456 return interpol;
457}

459double spline::deriv(int order, double x) const
460{
461 assert(order>0);
462 size_t n=m_x.size();
463 size_t idx = find_closest(x);
464
465 double h=x-m_x[idx];
466 double interpol;
467 if(x<m_x[0]) {
468 // extrapolation to the left
469 switch(order) {
470 case 1:
471 interpol=2.0*m_c0*h + m_b[0];
472 break;
473 case 2:
474 interpol=2.0*m_c0;
475 break;
476 default:
477 interpol=0.0;
478 break;
479 }
480 } else if(x>m_x[n-1]) {
481 // extrapolation to the right
482 switch(order) {
483 case 1:
484 interpol=2.0*m_c[n-1]*h + m_b[n-1];
485 break;
486 case 2:
487 interpol=2.0*m_c[n-1];
488 break;
489 default:
490 interpol=0.0;
491 break;
492 }
493 } else {
494 // interpolation
495 switch(order) {
496 case 1:
497 interpol=(3.0*m_d[idx]*h + 2.0*m_c[idx])*h + m_b[idx];
498 break;
499 case 2:
500 interpol=6.0*m_d[idx]*h + 2.0*m_c[idx];
501 break;
502 case 3:
503 interpol=6.0*m_d[idx];
504 break;
505 default:
506 interpol=0.0;
507 break;
508 }
509 }
510 return interpol;
511}

562// defines the new operator (), so that we can access the elements
563// by A(i,j), index going from i=0,...,dim()-1
564double & band_matrix::operator () (int i, int j)
565{
566 int k=j-i; // what band is the entry
567 assert( (i>=0) && (i<dim()) && (j>=0) && (j<dim()) );
568 assert( (-num_lower()<=k) && (k<=num_upper()) );
569 // k=0 -> diagonal, k<0 lower left part, k>0 upper right part
570 if(k>=0) return m_upper[k][i];
571 else return m_lower[-k][i];
572}

594// LR-Decomposition of a band matrix
595void band_matrix::lu_decompose()
596{
597 int i_max,j_max;
598 int j_min;
599 double x;
600
601 // preconditioning
602 // normalize column i so that a_ii=1
603 for(int i=0; i<this->dim(); i++) {
604 assert(this->operator()(i,i)!=0.0);
605 this->saved_diag(i)=1.0/this->operator()(i,i);
606 j_min=std::max(0,i-this->num_lower());
607 j_max=std::min(this->dim()-1,i+this->num_upper());
608 for(int j=j_min; j<=j_max; j++) {
609 this->operator()(i,j) *= this->saved_diag(i);
610 }
611 this->operator()(i,i)=1.0; // prevents rounding errors
612 }
613
614 // Gauss LR-Decomposition
615 for(int k=0; k<this->dim(); k++) {
616 i_max=std::min(this->dim()-1,k+this->num_lower()); // num_lower not a mistake!
617 for(int i=k+1; i<=i_max; i++) {
618 assert(this->operator()(k,k)!=0.0);
619 x=-this->operator()(i,k)/this->operator()(k,k);
620 this->operator()(i,k)=-x; // assembly part of L
621 j_max=std::min(this->dim()-1,k+this->num_upper());
622 for(int j=k+1; j<=j_max; j++) {
623 // assembly part of R
624 this->operator()(i,j)=this->operator()(i,j)+x*this->operator()(k,j);
625 }
626 }
627 }
628}
629// solves Ly=b
630std::vector<double> band_matrix::l_solve(const std::vector<double>& b) const
631{
632 assert( this->dim()==(int)b.size() );
633 std::vector<double> x(this->dim());
634 int j_start;
635 double sum;
636 for(int i=0; i<this->dim(); i++) {
637 sum=0;
638 j_start=std::max(0,i-this->num_lower());
639 for(int j=j_start; j<i; j++) sum += this->operator()(i,j)*x[j];
640 x[i]=(b[i]*this->saved_diag(i)) - sum;
641 }
642 return x;
643}
644// solves Rx=y
645std::vector<double> band_matrix::r_solve(const std::vector<double>& b) const
646{
647 assert( this->dim()==(int)b.size() );
648 std::vector<double> x(this->dim());
649 int j_stop;
650 double sum;
651 for(int i=this->dim()-1; i>=0; i--) {
652 sum=0;
653 j_stop=std::min(this->dim()-1,i+this->num_upper());
654 for(int j=i+1; j<=j_stop; j++) sum += this->operator()(i,j)*x[j];
655 x[i]=( b[i] - sum ) / this->operator()(i,i);
656 }
657 return x;
658}
659
660std::vector<double> band_matrix::lu_solve(const std::vector<double>& b,
661 bool is_lu_decomposed)
662{
663 assert( this->dim()==(int)b.size() );
664 std::vector<double> x,y;
665 if(is_lu_decomposed==false) {
666 this->lu_decompose();
667 }
668 y=this->l_solve(b);
669 x=this->r_solve(y);
670 return x;
671}
